\section{Bileşen 1: ApiKey Veri Modeli}
% ═══════════════════════════════════════════════════════════════════════════════

\subsection{Ürün Gereksinimi}

Kullanıcıya özgü, soft-delete destekli, son kullanım tarihi izlenebilir bir
API anahtarı tablosuna ihtiyaç vardı. Anahtarlar, hesap şifresiyle değil
bağımsız olarak iptal edilebilmelidir.

\subsection{Uygulama — Kod Entegrasyonu}

\textbf{Dosya:} \texttt{server/models.py}, satırlar 26--36.

\begin{lstlisting}[language=Python, caption={ApiKey Modeli — models.py}]
class ApiKey(db.Model):
    """VS Code Extension ve harici araçlar icin API anahtari."""
    id         = db.Column(db.Integer, primary_key=True)
    user_id    = db.Column(db.Integer, db.ForeignKey('user.id'),
                           nullable=False, index=True)
    name       = db.Column(db.String(100), nullable=False)  # e.g. "My VS Code"
    key        = db.Column(db.String(128), unique=True,
                           nullable=False, index=True)       # token degeri
    is_active  = db.Column(db.Boolean, default=True)        # soft-delete
    created_at = db.Column(db.DateTime, default=datetime.utcnow)
    last_used_at = db.Column(db.DateTime, nullable=True)    # izlenebilirlik

    user = db.relationship('User',
                           backref=db.backref('api_keys', lazy='dynamic'))
\end{lstlisting}

\subsection{Ürün Gereksinimi}

Kullanıcının kendi anahtarlarını listeleyebilmesi, yeni anahtar oluşturabilmesi

\subsection{API Sözleşmesi}

\begin{center}
\begin{tabular}{@{} P{2cm} P{4.5cm} P{7.5cm} @{}}
\toprule
\textbf{Metod} & \textbf{Endpoint} & \textbf{Açıklama} \\
\midrule
\texttt{GET} & \texttt{/api/keys} &
  Kullanıcıya ait aktif anahtarları listele.
  Token değeri gizlenerek yalnızca son 4 karakter gösterilir
  (\texttt{ca-***-xxxx}). \\[6pt]
\texttt{POST} & \texttt{/api/keys} &

\texttt{DELETE} & \texttt{/api/keys/<id>} &

  Fiziksel silme yapılmaz. \\
\bottomrule
\end{tabular}
\end{center}

\subsection{Kod Entegrasyon Noktaları}

\textbf{Dosya:} \texttt{server/app.py}, satırlar 6769--6825.

\begin{lstlisting}[language=Python, caption={GET /api/keys — Anahtar Listeleme}]
@app.route('/api/keys', methods=['GET'])
@jwt_required()
def get_api_keys():
    user_id = get_jwt_identity()
    keys = ApiKey.query.filter_by(
        user_id=user_id, is_active=True
    ).order_by(ApiKey.created_at.desc()).all()
    return jsonify({
        'keys': [{
            'id': k.id,
            'name': k.name,
            # Guvenlik: son 4 karakter haric gosterilmez
            'key_preview': f"ca-***-{k.key[-4:]}",
            'created_at': k.created_at.isoformat(),
            'last_used_at': k.last_used_at.isoformat()
                             if k.last_used_at else None
        } for k in keys]
    })
\end{lstlisting}

\begin{lstlisting}[language=Python, caption={POST /api/keys — Anahtar Oluşturma}]
@app.route('/api/keys', methods=['POST'])
@jwt_required()
def create_api_key():
    user_id = get_jwt_identity()
    data    = request.get_json() or {}
    name    = data.get('name', 'My API Key').strip()

    # "ca-" on eki ile benzersiz token uret
    token   = f"ca-{secrets.token_hex(16)}"

    new_key = ApiKey(user_id=user_id, name=name, key=token)
    db.session.add(new_key)
    db.session.commit()

    # Tam token YALNIZCA burada donus yapar
    return jsonify({'key': {'id': new_key.id, 'token': token}})
\end{lstlisting}

\begin{lstlisting}[language=Python, caption={DELETE /api/keys/<id> — Anahtar İptali}]
@app.route('/api/keys/<int:key_id>', methods=['DELETE'])
@jwt_required()
def revoke_api_key(key_id):
    user_id    = get_jwt_identity()
    key_record = ApiKey.query.filter_by(
        id=key_id, user_id=user_id
    ).first()
    if not key_record:
        return jsonify({'error': 'Key not found'}), 404
    key_record.is_active = False   # Soft-delete
    db.session.commit()
    return jsonify({'message': 'Key revoked successfully'})
\end{lstlisting}

\begin{center}

\section{Bileşen 3: /v1/ask Harici Köprü Endpoint'i}
% ═══════════════════════════════════════════════════════════════════════════════

\subsection{Ürün Gereksinimi}

VS Code eklentisi, terminal script veya CI/CD pipeline gibi araçların
JWT oturumu gerektirmeden, yalnızca bir API anahtarı ile CodeAlchemist'e
soru sorabilmesi gerekir.

\subsection{API Sözleşmesi}

\begin{tcolorbox}[codebox, title={Endpoint: POST /v1/ask}]
\textbf{Header:}\\
\hspace*{4mm}\texttt{X-API-Key: ca-<token>}\\
\hspace*{4mm}\texttt{Content-Type: application/json}\\[4pt]
\textbf{Body (JSON):}\\
\hspace*{4mm}\texttt{\{"question": "...", "code": "..."\}}\\[4pt]
\textbf{Başarılı Yanıt (200):}\\
\hspace*{4mm}\texttt{\{"answer": "<AI yaniti>"\}}\\[4pt]
\textbf{Hata Yanıtları:}\\
\hspace*{4mm}\texttt{401} — Header eksik veya anahtar geçersiz/iptal edilmiş.\\
\hspace*{4mm}\texttt{400} — \texttt{question} ve \texttt{code} alanları birden boş.
\end{tcolorbox}

\subsection{Kod Entegrasyonu}

\textbf{Dosya:} \texttt{server/app.py}, satırlar 6828--6888.

\begin{lstlisting}[language=Python, caption={/v1/ask — Harici Koprosu Endpoint}]
@app.route('/v1/ask', methods=['POST'])
def external_ask():
    # 1. Kimlik Dogrulama
    api_key_header = request.headers.get('X-API-Key')
    if not api_key_header:
        return jsonify({'error': 'X-API-Key header is missing'}), 401

    key_record = ApiKey.query.filter_by(
        key=api_key_header, is_active=True
    ).first()
    if not key_record:
        return jsonify({'error': 'Invalid or revoked API Key'}), 401

    # 2. Izlenebilirlik: Son kullanim tarihini guncelle
    key_record.last_used_at = datetime.datetime.utcnow()
    db.session.commit()

    # 3. Istegi isle
    data     = request.get_json(silent=True) or {}
    question = data.get('question', '')
    code     = data.get('code', '')

    if not question and not code:
        return jsonify({'error': 'question or code required'}), 400

    # 4. Kullanici tercihlerini al (kisisellestirme)
    user  = User.query.get(key_record.user_id)
    prefs = json.loads(user.preferences) if user and user.preferences else {}

    # 5. AI yaniti uret (Gemini fallback zinciri)
    full_text = ""
    for chunk in generate_gemini_answer(question=question,
                                        code=code, prefs=prefs):
        full_text += chunk

    return jsonify({'answer': full_text})
\end{lstlisting}

\subsection{Streaming Desteği}

Endpoint, \texttt{Accept: text/event-stream} header'ı gönderildiğinde
Server-Sent Events (SSE) ile akış desteği sunar:

\begin{lstlisting}[language=Python, caption={SSE Stream Modu}]
if request.headers.get('Accept') == 'text/event-stream':
    def generate():
        for chunk in generator:
            yield f"data: {json.dumps({'text': chunk})}\n\n"
        yield "data: [DONE]\n\n"
    return Response(stream_with_context(generate()),
                    mimetype='text/event-stream')
\end{lstlisting}

\begin{tcolorbox}[infobox, title={Tasarım Notu}]

Böylece CI/CD pipeline veya cron job gibi oturum kavramı olmayan
ortamlar JWT olmadan çağrı yapabilir.
\end{tcolorbox}

\subsection{Kullanıcı Tercihleri Entegrasyonu}

const askCmd = vscode.commands.registerCommand(
  'codeAlchemist.ask', async () => {
    const cfg      = vscode.workspace.getConfiguration('codeAlchemist');
    const apiKey   = (cfg.get<string>('apiKey') || '').trim();
    const endpoint = (cfg.get<string>('endpoint') || '').trim();

    if (!apiKey)   { /* hata mesaji */ return; }
    if (!endpoint) { /* hata mesaji */ return; }

    // Secili kodu al
    const editor = vscode.window.activeTextEditor;
    const selectedText = editor && !editor.selection.isEmpty
      ? editor.document.getText(editor.selection) : '';

    const question = await vscode.window.showInputBox({
      title: 'Ask CodeAlchemist',
      prompt: selectedText
        ? 'Question (selected code will be included)'
        : 'Question',
    });
    if (!question) return;

    output.show(true);
    output.appendLine(`POST ${endpoint}`);
    output.appendLine(`Question: ${question}`);
    if (selectedText) {
      output.appendLine('--- Selected code ---');
      output.appendLine(selectedText);
      output.appendLine('---');
    }
    output.appendLine('Waiting for response...');

    // API cagris
    const res  = await fetch(endpoint, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json',
                 'X-API-Key': apiKey },
      body: JSON.stringify({ question, code: selectedText }),
    });

    const data = await res.json() as { answer?: string };
    output.appendLine('');
    output.appendLine(data.answer ?? 'No answer received.');
  }
);
\end{lstlisting}

                bonus_pct=pkg_data.get('bonus_pct', 0),
                is_active=True,
            )
            db.session.add(pkg)
        db.session.commit()
        print(f"Seeded {len(DEFAULT_TOKEN_PACKAGES)} token packages.")

\subsection{Checkout Session Oluşturma — Kod Entegrasyonu}

\textbf{Dosya:} \texttt{server/app.py} — \texttt{POST /api/billing/checkout-session}.

\begin{lstlisting}[language=Python, caption={Checkout Session — Stripe API cagrisi ve DB kaydi}]
@app.route('/api/billing/checkout-session', methods=['POST'])
@jwt_required()
def create_checkout_session():
    if not _STRIPE_SECRET_KEY:
        return jsonify({'error': 'Billing not configured.'}), 503

    user_id    = get_jwt_identity()
    package_id = (request.get_json(silent=True) or {}).get('package_id')

    # Paketi bul (int DB id veya string slug ile)
    pkg = TokenPackage.query.filter_by(
        id=int(package_id), is_active=True
    ).first()

    bonus_tokens = int(pkg.tokens * pkg.bonus_pct / 100) if pkg.bonus_pct else 0
    total_tokens = pkg.tokens + bonus_tokens
    price_cents  = int(round(pkg.price_usd * 100))

    host        = request.host_url.rstrip('/')
    success_url = f"{host}/?payment=success&tokens={total_tokens}"
    cancel_url  = f"{host}/?payment=canceled"

    try:
        session = stripe.checkout.Session.create(
            payment_method_types=['card'],
            mode='payment',
            customer_email=User.query.get(user_id).email,
            line_items=[{
                'price_data': {
                    'currency': 'usd',
                    'unit_amount': price_cents,
                    'product_data': {
                        'name': f"{pkg.name} -- {total_tokens} Tokens"
                    },
                },
                'quantity': 1,
            }],
            success_url=success_url,
            cancel_url=cancel_url,
            metadata={
                'user_id': str(user_id),
                'tokens_granted': str(total_tokens),
            },
        )
    except stripe.error.StripeError as e:
        return jsonify({'error': str(e)}), 502

    # Pending kayit
    purchase = TokenPurchase(
        user_id=user_id, tokens_granted=total_tokens,
        amount_cents=price_cents, currency='usd',
        stripe_checkout_session_id=session.id, status='pending',
    )
    db.session.add(purchase)
    db.session.commit()

    return jsonify({'checkout_url': session.url})
\end{lstlisting}

\subsection{Webhook Handler — Idempotency ile Fulfillment}

\textbf{Dosya:} \texttt{server/app.py} — \texttt{POST /api/billing/webhook}.

\begin{lstlisting}[language=Python, caption={Stripe Webhook — Idempotency kontrolu ve token yukleme}]
@app.route('/api/billing/webhook', methods=['POST'])
def stripe_webhook():
    payload    = request.get_data(as_text=False)
    sig_header = request.headers.get('Stripe-Signature', '')

    try:
        event = stripe.Webhook.construct_event(
            payload, sig_header, _STRIPE_WEBHOOK_SECRET
        )
    except stripe.error.SignatureVerificationError:
        return jsonify({'error': 'Invalid signature'}), 400

    if event['type'] == 'checkout.session.completed':
        session_obj       = event['data']['object']
        stripe_session_id = session_obj.get('id')

        purchase = TokenPurchase.query.filter_by(
            stripe_checkout_session_id=stripe_session_id
        ).first()

        # --- Idempotency: Zaten tamamlandi mi? ---
        if purchase and purchase.status == 'completed':
            return jsonify({'received': True}), 200   # Tekrar isleme

        # TokenBalance guncelle
        balance_record = TokenBalance.query.filter_by(
            user_id=purchase.user_id
        ).first()
        balance_record.balance += purchase.tokens_granted

        # Transaction kaydet
        db.session.add(TokenTransaction(
            user_id=purchase.user_id,
            amount=purchase.tokens_granted,
            type='purchase',
            description=f"Token purchase: {purchase.package_name}",
            reference_id=stripe_session_id,
        ))

        # Satinalma kaydi tamamlandi
        purchase.status       = 'completed'
        purchase.completed_at = datetime.datetime.utcnow()
        db.session.commit()

    return jsonify({'received': True}), 200
\end{lstlisting}
